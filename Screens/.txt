  const [text, setText] = useState("");

  // ✅ Create the Google Auth request
  const [request, response, promptAsync] = Google.useAuthRequest({
    expoClientId: "1095714555219-p555htrn8rm7ii6m4icpmr6rqikgcmhg.apps.googleusercontent.com", // Web client ID
    androidClientId: "1095714555219-2qqj58v0oiiv9dvfr2jq4mnqpp923b32.apps.googleusercontent.com",
    scopes: ["profile", "email"],
  });

  // ✅ Handle the Google sign-in response
  useEffect(() => {
    if (response?.type === "success") {
      const { authentication } = response;
      console.log("✅ Google Login Success");
      getUserInfo(authentication.accessToken);
    } else if (response?.type === "error") {
      console.error("❌ Google Login Error:", response.error);
    }
  }, [response]);

  // ✅ Fetch user info from Google API
  const getUserInfo = async (token) => {
    try {
      const res = await axios.get("https://www.googleapis.com/userinfo/v2/me", {
        headers: { Authorization: `Bearer ${token}` },
      });

      console.log("👤 User Info:", res.data);

      // Save user data locally
      await AsyncStorage.setItem("@user", JSON.stringify(res.data));

      // ✅ Navigate to MainTabs
      navigation.reset({
        index: 0,
        routes: [{ name: "MainTabs" }],
      });
    } catch (err) {
      console.error("❌ Error fetching user info:", err);
    }
  };

  return (
    <View style={styles.container}>
      <Text style={styles.logo}>💎</Text>
      <Text style={styles.title}>Hello, Gorgeous</Text>

      <Button
        title="Login with Google"
        onPress={() => {
          if (request) {
            // ✅ FIX: useProxy ensures redirect URI goes through Expo, avoiding Error 400
            promptAsync({ useProxy: true });
          } else {
            console.log("⚠️ Google request not ready yet");
          }
        }}
      />

      <Text style={styles.footer}>
        By continuing, you agree to our{" "}
        <Text style={styles.link}>Terms</Text> &{" "}
        <Text style={styles.link}>Privacy Policy</Text>
      </Text>
    </View>
  );
}
