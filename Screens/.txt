import React, { useEffect, useState } from "react";
import { View, Text, StyleSheet, Button } from "react-native";
import * as WebBrowser from "expo-web-browser";
import * as Google from "expo-auth-session/providers/google";
import AsyncStorage from "@react-native-async-storage/async-storage";
import axios from "axios";

WebBrowser.maybeCompleteAuthSession();

export default function Auth({ navigation }) {
  const [text, setText] = useState("");

  // ‚úÖ Create the Google Auth request
  const [request, response, promptAsync] = Google.useAuthRequest({
    expoClientId: "1095714555219-p555htrn8rm7ii6m4icpmr6rqikgcmhg.apps.googleusercontent.com", // Web client ID
    androidClientId: "1095714555219-2qqj58v0oiiv9dvfr2jq4mnqpp923b32.apps.googleusercontent.com",
    scopes: ["profile", "email"],
  });

  // ‚úÖ Handle the Google sign-in response
  useEffect(() => {
    if (response?.type === "success") {
      const { authentication } = response;
      console.log("‚úÖ Google Login Success");
      getUserInfo(authentication.accessToken);
    } else if (response?.type === "error") {
      console.error("‚ùå Google Login Error:", response.error);
    }
  }, [response]);

  // ‚úÖ Fetch user info from Google API
  const getUserInfo = async (token) => {
    try {
      const res = await axios.get("https://www.googleapis.com/userinfo/v2/me", {
        headers: { Authorization: `Bearer ${token}` },
      });

      console.log("üë§ User Info:", res.data);

      // Save user data locally
      await AsyncStorage.setItem("@user", JSON.stringify(res.data));

      // ‚úÖ Navigate to MainTabs
      navigation.reset({
        index: 0,
        routes: [{ name: "MainTabs" }],
      });
    } catch (err) {
      console.error("‚ùå Error fetching user info:", err);
    }
  };

  return (
    <View style={styles.container}>
      <Text style={styles.logo}>üíé</Text>
      <Text style={styles.title}>Hello, Gorgeous</Text>

      <Button
        title="Login with Google"
        onPress={() => {
          if (request) {
            // ‚úÖ FIX: useProxy ensures redirect URI goes through Expo, avoiding Error 400
            promptAsync({ useProxy: true });
          } else {
            console.log("‚ö†Ô∏è Google request not ready yet");
          }
        }}
      />

      <Text style={styles.footer}>
        By continuing, you agree to our{" "}
        <Text style={styles.link}>Terms</Text> &{" "}
        <Text style={styles.link}>Privacy Policy</Text>
      </Text>
    </View>
  );
}
